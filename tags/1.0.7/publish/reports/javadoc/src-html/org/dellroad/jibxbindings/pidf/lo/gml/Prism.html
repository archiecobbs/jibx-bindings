008    package org.dellroad.jibxbindings.pidf.lo.gml;
009    
010    /**
011     * GML prism.
012     */
013    public class Prism extends Solid {
014    
015        private Surface base;
016        private Measure height;
017    
018        /**
019         * The prism's {@code <gs:base>}.
020         */
021        public Surface getBase() {
022            return this.base;
023        }
024        public void setBase(Surface base) {
025            this.base = base;
026        }
027    
028        /**
029         * The prism's {@code <gs:height>}.
030         */
031        public Measure getHeight() {
032            return this.height;
033        }
034        public void setHeight(Measure height) {
035            this.height = height;
036        }
037    
038        @Override
039        public void visit(GMLObjectSwitch gmlObjectSwitch) {
040            gmlObjectSwitch.casePrism(this);
041        }
042    }
