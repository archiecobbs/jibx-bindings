010import java.util.Date;
011
012/**
013 * Represents a <code>&lt;Call&gt;</code>.
014 */
015public class Call extends SidAccountResource implements TwilioResponseContent {

017    private String parentCallSid;
018    private String to;
019    private String from;
020    private String phoneNumberSid;
021    private CallStatus status;
022    private Date startTime;
023    private Date endTime;
024    private int duration;
025    private double price;
026    private String priceUnit;

028    private AnswerType answeredBy;

033    public String getParentCallSid() {

037        this.parentCallSid = parentCallSid;
038    }
039
040    public String getTo() {
041        return this.to;
042    }
043    public void setTo(String to) {
044        this.to = to;
045    }
046
047    public String getFrom() {
048        return this.from;
049    }
050    public void setFrom(String from) {
051        this.from = from;
052    }
053
054    public String getPhoneNumberSid() {
055        return this.phoneNumberSid;
056    }
057    public void setPhoneNumberSid(String phoneNumberSid) {
058        this.phoneNumberSid = phoneNumberSid;
059    }
060
061    public CallStatus getStatus() {
062        return this.status;
063    }
064    public void setStatus(CallStatus status) {
065        this.status = status;
066    }
067
068    public Date getStartTime() {

076        return this.endTime;

078    public void setEndTime(Date endTime) {
079        this.endTime = endTime;
080    }

084    }

087    }
088
089    public double getPrice() {

096    public String getPriceUnit() {

101    }
102
103    public CallDirection getDirection() {
104        return this.direction;
105    }
106    public void setDirection(CallDirection direction) {
107        this.direction = direction;
108    }
109
110    public AnswerType getAnsweredBy() {
111        return this.answeredBy;
112    }
113    public void setAnsweredBy(AnswerType answeredBy) {
114        this.answeredBy = answeredBy;
115    }
116
117    public String getForwardedFrom() {
118        return this.forwardedFrom;
119    }
120    public void setForwardedFrom(String forwardedFrom) {
121        this.forwardedFrom = forwardedFrom;
122    }
123
124    public String getCallerName() {
125        return this.callerName;
126    }
127    public void setCallerName(String callerName) {
128        this.callerName = callerName;
129    }
130
131    public APIVersion getAPIVersion() {
132        return this.apiVersion;
133    }
134    public void setAPIVersion(APIVersion apiVersion) {
135        this.apiVersion = apiVersion;
136    }
