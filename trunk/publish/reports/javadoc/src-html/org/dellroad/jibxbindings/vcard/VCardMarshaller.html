008package org.dellroad.jibxbindings.vcard;
009
010import java.io.IOException;
011
012import javax.xml.parsers.DocumentBuilderFactory;
013import javax.xml.parsers.ParserConfigurationException;
014
015import org.jibx.extras.DomElementMapper;
016import org.jibx.runtime.IMarshallingContext;
017import org.jibx.runtime.IUnmarshallingContext;
018import org.jibx.runtime.JiBXException;
019import org.w3c.dom.Document;
020import org.w3c.dom.Element;
021
022import ezvcard.VCard;
023import ezvcard.io.xml.XCardDocument;
024
025/**
026 * JiBX marshaller/unmarshaller for a single {@link VCard}.
027 *
028 * @see <a href="https://ez-vcard.googlecode.com">The EZ-vCard Project</a>
029 * @see <a href="https://tools.ietf.org/html/rfc6351">RFC 6351</a>
030 */
031public class VCardMarshaller extends DomElementMapper {
032
033    public static final String VCARD_NAMESPACE_URI = "urn:ietf:params:xml:ns:vcard-4.0";
034    public static final String VCARDS_ELEMENT_NAME = "vcards";
035    public static final String VCARD_ELEMENT_NAME = "vcard";
036
037    public VCardMarshaller() throws JiBXException {
038        this(VCARD_NAMESPACE_URI, 0, VCARD_ELEMENT_NAME);
039    }
040
041    public VCardMarshaller(String uri, int index, String name) throws JiBXException {
042        super(uri, index, name);
043    }
044
045    @Override
046    public void marshal(Object obj, IMarshallingContext ctx) throws JiBXException {
047
048        // Handle null
049        if (obj == null)
050            return;
051
052        // Convert to XML
053        final XCardDocument xcard = new XCardDocument();
054        xcard.add((VCard)obj);
055        super.marshal(xcard.getDocument().getDocumentElement().getFirstChild(), ctx);
056    }
057
058    @Override
059    public VCard unmarshal(Object obj, IUnmarshallingContext ctx) throws JiBXException {
060
061        // Get XML element
062        final Element element = (Element)super.unmarshal(obj, ctx);
063
064        // Handle null
065        if (element == null)
066            return null;
067
068        // Create temporary XML <vcards> document
069        final Document doc;
070        try {
071            doc = DocumentBuilderFactory.newInstance().newDocumentBuilder().newDocument();
072        } catch (ParserConfigurationException e) {
073            throw new RuntimeException("unexpected exception", e);
074        }
075        final Element vcards = doc.createElementNS(VCARD_NAMESPACE_URI, VCARDS_ELEMENT_NAME);
076        doc.appendChild(vcards);
077
078        // Append <vcard> node
079        vcards.appendChild(doc.importNode(element, true));
080
081        // Read it
082        try {
083            return new XCardDocument(doc).reader().readNext();
084        } catch (IOException e) {
085            throw new RuntimeException("unexpected exception", e);
086        }
087    }
088
089    /**
090     * Clone a {@link VCard} by converting it to XML and back.
091     * This method works around the fact that {@link VCard} does not implement {@link Cloneable}.
092     *
093     * @throws IllegalArgumentException if {@code vcard} is null
094     */
095    public static VCard clone(VCard vcard) {
096
097        // Sanity check
098        if (vcard == null)
099            throw new IllegalArgumentException("null vcard");
100
101        // Convert to XML
102        final XCardDocument xcard = new XCardDocument();
103        xcard.add(vcard);
104        final Document doc = xcard.getDocument();
105
106        // Convert from XML
107        try {
108            return new XCardDocument(doc).reader().readNext();
109        } catch (IOException e) {
110            throw new RuntimeException("unexpected exception", e);
111        }
112    }
113}
