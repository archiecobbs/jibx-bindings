010    /**
011     * Implemented by {@link Enum}s whose values' XML representations are different from their {@link Enum#name names}.
012     * To make use of this interface, specify <code>enum-value-method="getXMLName"</code> in the JiBX binding file.
013     */
014    public interface XMLEnum {
015    
016        /**
017         * Get the unique string to use to represent this instance in an XML file.
018         */
019        String getXMLName();
020    }
