008    package org.dellroad.jibxbindings.twilio.restapi;
009    
010    import org.dellroad.jibxbindings.ParseUtil;
011    import org.dellroad.jibxbindings.XMLEnum;
012    import org.jibx.runtime.JiBXParseException;
013    
014    /**
015     * SMS directionality.
016     */
017    public enum SMSDirection implements XMLEnum {
018    
019        /**
020         * Inbound SMS.
021         */
022        INBOUND("inbound", false),
023    
024        /**
025         * Outbound, initiated via the REST API.
026         */
027        OUTBOUND_API("outbound-api", true),
028    
029        /**
030         * Outbound, initiated during a call.
031         */
032        OUTBOUND_CALL("outbound-call", true),
033    
034        /**
035         * Outbound, initiated in response to an incoming SMS.
036         */
037        OUTBOUND_REPLY("outbound-reply", true);
038    
039        private final String xmlName;
040        private final boolean outbound;
041    
042        private SMSDirection(String xmlName, boolean outbound) {
043            this.xmlName = xmlName;
044            this.outbound = outbound;
045        }
046    
047        /**
048         * Is the SMS inbound or outbound?
049         *
050         * @return false for {@link #INBOUND}, true for all other values
051         */
052        public boolean isOutbound() {
053            return this.outbound;
054        }
055    
056        @Override
057        public String getXMLName() {
058            return this.xmlName;
059        }
060    
061        public static SMSDirection deserializeXML(String string) throws JiBXParseException {
062            return ParseUtil.deserializeXMLEnum(string, SMSDirection.values());
063        }
064    }
