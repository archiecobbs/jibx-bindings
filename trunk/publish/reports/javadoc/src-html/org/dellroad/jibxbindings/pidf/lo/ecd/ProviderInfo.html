008package org.dellroad.jibxbindings.pidf.lo.ecd;
009
010import java.net.URI;
011import java.util.ArrayList;
012import java.util.List;
013
014import ezvcard.VCard;
015
016/**
017 * The {@code <pi:EmergencyCallData.ProviderInfo>} element.
018 */
019public class ProviderInfo {
020
021    private String dataProviderReference;                   // mandatory
022    private String dataProviderString;                      // mandatory
023    private String providerId;
024    private String providerIdSeries;
025    private String typeOfProvider;
026    private URI contactURI;                                 // mandatory
027    private List<String> languages = new ArrayList<>();     // mandatory (one or more)
028    private VCard dataProviderContact;
029    private String subcontratorPrincipal;
030    private SubcontractorPriority subcontratorPriority;
031
032    public String getDataProviderReference() {
033        return this.dataProviderReference;
034    }
035    public void setDataProviderReference(String dataProviderReference) {
036        this.dataProviderReference = dataProviderReference;
037    }
038
039    public String getDataProviderString() {
040        return this.dataProviderString;
041    }
042    public void setDataProviderString(String dataProviderString) {
043        this.dataProviderString = dataProviderString;
044    }
045
046    public String getProviderIdSeries() {
047        return this.providerIdSeries;
048    }
049    public void setProviderIdSeries(String providerIdSeries) {
050        this.providerIdSeries = providerIdSeries;
051    }
052
053    public String getTypeOfProvider() {
054        return this.typeOfProvider;
055    }
056    public void setTypeOfProvider(String typeOfProvider) {
057        this.typeOfProvider = typeOfProvider;
058    }
059
060    public URI getContactURI() {
061        return this.contactURI;
062    }
063    public void setContactURI(URI contactURI) {
064        this.contactURI = contactURI;
065    }
066
067    public List<String> getLanguages() {
068        return this.languages;
069    }
070    public void setLanguages(List<String> languages) {
071        this.languages = languages;
072    }
073
074    public VCard getDataProviderContact() {
075        return this.dataProviderContact;
076    }
077    public void setDataProviderContact(VCard dataProviderContact) {
078        this.dataProviderContact = dataProviderContact;
079    }
080
081    public String getSubcontratorPrincipal() {
082        return this.subcontratorPrincipal;
083    }
084    public void setSubcontratorPrincipal(String subcontratorPrincipal) {
085        this.subcontratorPrincipal = subcontratorPrincipal;
086    }
087
088    public SubcontractorPriority getSubcontratorPriority() {
089        return this.subcontratorPriority;
090    }
091    public void setSubcontratorPriority(SubcontractorPriority subcontratorPriority) {
092        this.subcontratorPriority = subcontratorPriority;
093    }
094
095// JiBX
096
097    public boolean hasDataProviderContact() {
098        return this.getDataProviderContact() != null;
099    }
100}
