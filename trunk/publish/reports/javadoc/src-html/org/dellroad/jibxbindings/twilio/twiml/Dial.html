008package org.dellroad.jibxbindings.twilio.twiml;
009
010import java.net.URI;
011import java.util.ArrayList;
012import java.util.List;
013
014import org.dellroad.jibxbindings.twilio.Method;
015
016/**
017 * TwiML <code>&lt;Dial&gt;</code> verb.
018 */
019public class Dial implements Verb {
020
021    public static final int DEFAULT_TIMEOUT = 30;
022    public static final int DEFAULT_TIME_LIMIT = 4 * 60 * 60;
023
024    private URI action;
025    private Method method = Method.POST;
026    private int timeout = DEFAULT_TIMEOUT;
027    private boolean hangupOnStar;
028    private int timeLimit = DEFAULT_TIME_LIMIT;
029    private String callerId;
030    private List<Number> numbers = new ArrayList<Number>();
031    private Conference conference;
032
033    public URI getAction() {
034        return this.action;
035    }
036    public void setAction(URI action) {
037        this.action = action;
038    }
039
040    public Method getMethod() {
041        return this.method;
042    }
043    public void setMethod(Method method) {
044        this.method = method;
045    }
046
047    public int getTimeout() {
048        return this.timeout;
049    }
050    public void setTimeout(int timeout) {
051        this.timeout = timeout;
052    }
053
054    public boolean isHangupOnStar() {
055        return this.hangupOnStar;
056    }
057    public void setHangupOnStar(boolean hangupOnStar) {
058        this.hangupOnStar = hangupOnStar;
059    }
060
061    public int getTimeLimit() {
062        return this.timeLimit;
063    }
064    public void setTimeLimit(int timeLimit) {
065        this.timeLimit = timeLimit;
066    }
067
068    public String getCallerId() {
069        return this.callerId;
070    }
071    public void setCallerId(String callerId) {
072        this.callerId = callerId;
073    }
074
075    public List<Number> getNumbers() {
076        return this.numbers;
077    }
078    public void setNumbers(List<Number> numbers) {
079        this.numbers = numbers;
080    }
081
082    public Conference getConference() {
083        return this.conference;
084    }
085    public void setConference(Conference conference) {
086        this.conference = conference;
087    }
088}
