008package org.dellroad.jibxbindings.pidf.lo.ecd;
009
010import java.util.ArrayList;
011import java.util.Iterator;
012import java.util.List;
013
014/**
015 * The {@code <gp:EmergencyCallDataValue>} element.
016 */
017public class EmergencyCallDataValue implements Cloneable {
018
019    private List<ProviderInfo> providerInfos = new ArrayList<>();
020    private List<ServiceInfo> serviceInfos = new ArrayList<>();
021    private List<DeviceInfo> deviceInfos = new ArrayList<>();
022    private List<SubscriberInfo> subscriberInfos = new ArrayList<>();
023    private List<Comment> comments = new ArrayList<>();
024
025    public List<ProviderInfo> getProviderInfos() {
026        return this.providerInfos;
027    }
028    public void setProviderInfos(List<ProviderInfo> providerInfos) {
029        this.providerInfos = providerInfos;
030    }
031
032    public List<ServiceInfo> getServiceInfos() {
033        return this.serviceInfos;
034    }
035    public void setServiceInfos(List<ServiceInfo> serviceInfos) {
036        this.serviceInfos = serviceInfos;
037    }
038
039    public List<DeviceInfo> getDeviceInfos() {
040        return this.deviceInfos;
041    }
042    public void setDeviceInfos(List<DeviceInfo> deviceInfos) {
043        this.deviceInfos = deviceInfos;
044    }
045
046    public List<SubscriberInfo> getSubscriberInfos() {
047        return this.subscriberInfos;
048    }
049    public void setSubscriberInfos(List<SubscriberInfo> subscriberInfos) {
050        this.subscriberInfos = subscriberInfos;
051    }
052
053    public List<Comment> getComments() {
054        return this.comments;
055    }
056    public void setComments(List<Comment> comments) {
057        this.comments = comments;
058    }
059
060// JiBX
061
062    public Iterator<AbstractDataProviderReferencing> iterateInfos() {
063        final ArrayList<AbstractDataProviderReferencing> list = new ArrayList<>();
064        list.addAll(this.providerInfos);
065        list.addAll(this.serviceInfos);
066        list.addAll(this.deviceInfos);
067        list.addAll(this.subscriberInfos);
068        list.addAll(this.comments);
069        return list.iterator();
070    }
071
072    public void addInfo(AbstractDataProviderReferencing info) {
073        if (info instanceof ProviderInfo)
074            this.providerInfos.add((ProviderInfo)info);
075        if (info instanceof ServiceInfo)
076            this.serviceInfos.add((ServiceInfo)info);
077        if (info instanceof DeviceInfo)
078            this.deviceInfos.add((DeviceInfo)info);
079        if (info instanceof SubscriberInfo)
080            this.subscriberInfos.add((SubscriberInfo)info);
081        if (info instanceof Comment)
082            this.comments.add((Comment)info);
083    }
084
085// Cloneable
086
087    @Override
088    public EmergencyCallDataValue clone() {
089        final EmergencyCallDataValue clone;
090        try {
091            clone = (EmergencyCallDataValue)super.clone();
092        } catch (CloneNotSupportedException e) {
093            throw new RuntimeException(e);
094        }
095        if (this.providerInfos != null) {
096            clone.providerInfos = new ArrayList<>(this.providerInfos.size());
097            for (ProviderInfo providerInfo : this.providerInfos)
098                clone.providerInfos.add(providerInfo.clone());
099        }
100        if (this.serviceInfos != null) {
101            clone.serviceInfos = new ArrayList<>(this.serviceInfos.size());
102            for (ServiceInfo serviceInfo : this.serviceInfos)
103                clone.serviceInfos.add(serviceInfo.clone());
104        }
105        if (this.deviceInfos != null) {
106            clone.deviceInfos = new ArrayList<>(this.deviceInfos.size());
107            for (DeviceInfo deviceInfo : this.deviceInfos)
108                clone.deviceInfos.add(deviceInfo.clone());
109        }
110        if (this.subscriberInfos != null) {
111            clone.subscriberInfos = new ArrayList<>(this.subscriberInfos.size());
112            for (SubscriberInfo subscriberInfo : this.subscriberInfos)
113                clone.subscriberInfos.add(subscriberInfo.clone());
114        }
115        if (this.comments != null) {
116            clone.comments = new ArrayList<>(this.comments.size());
117            for (Comment comment : this.comments)
118                clone.comments.add(comment.clone());
119        }
120        return clone;
121    }
122}
