008    package org.dellroad.jibxbindings.twilio.restapi;
009    
010    /**
011     * Abstract superclass for XML represenations of REST resources associated with a specific account.
012     */
013    public abstract class AccountResource extends Resource {
014    
015        private String accountSid;
016    
017        /**
018         * Get the unique ID of the account associated with this resource.
019         */
020        public String getAccountSid() {
021            return this.accountSid;
022        }
023        public void setAccountSid(String accountSid) {
024            this.accountSid = accountSid;
025        }
026    }
