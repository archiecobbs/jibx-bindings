008package org.dellroad.jibxbindings.pidf.held;
009
010import java.util.ArrayList;
011import java.util.List;
012
013/**
014 * The {@code <held:locationType>} XML element.
015 */
016public class LocationType {
017
018    /**
019     * Value for the {@link #types} field meaning any type of response is acceptable.
020     */
021    public static final String ANY = "any";
022
023    /**
024     * Value for the {@link #types} field meaning civic address.
025     */
026    public static final String CIVIC = "civic";
027
028    /**
029     * Value for the {@link #types} field meaning geodetic region.
030     */
031    public static final String GEODETIC = "geodetic";
032
033    /**
034     * Value for the {@link #types} field meaning a dereferencable URI to the actual location.
035     */
036    public static final String LOCATION_URI = "locationURI";
037
038    private boolean exact;
039    private List<String> types = new ArrayList<String>();
040
041    public boolean isExact() {
042        return this.exact;
043    }
044    public void setExact(boolean exact) {
045        this.exact = exact;
046    }
047
048    public List<String> getTypes() {
049        return this.types;
050    }
051    public void setTypes(List<String> types) {
052        this.types = types;
053    }
054}
