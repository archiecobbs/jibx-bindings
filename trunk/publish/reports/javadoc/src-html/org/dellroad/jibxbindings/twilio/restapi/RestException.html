008package org.dellroad.jibxbindings.twilio.restapi;
009
010import java.net.URI;
011
012/**
013 * Represents a <code>&lt;RestException&gt;</code>.
014 */
015public class RestException implements TwilioResponseContent {
016
017    private int status;
018    private String message;
019    private int code;
020    private URI moreInfo;
021
022    public int getStatus() {
023        return this.status;
024    }
025    public void setStatus(int status) {
026        this.status = status;
027    }
028
029    public String getMessage() {
030        return this.message;
031    }
032    public void setMessage(String message) {
033        this.message = message;
034    }
035
036    public int getCode() {
037        return this.code;
038    }
039    public void setCode(int code) {
040        this.code = code;
041    }
042
043    public URI getMoreInfo() {
044        return this.moreInfo;
045    }
046    public void setMoreInfo(URI moreInfo) {
047        this.moreInfo = moreInfo;
048    }
049}
