008    package org.dellroad.jibxbindings.twilio.restapi;
009    
010    /**
011     * Represents an <code>&lt;Participant&gt;</code>.
012     */
013    public class Participant extends AccountResource implements TwilioResponseContent {
014    
015        private String callSid;
016        private String conferenceSid;
017        private boolean muted;
018        private boolean startConferenceOnEnter;
019        private boolean endConferenceOnExit;
020    
021        public String getCallSid() {
022            return this.callSid;
023        }
024        public void setCallSid(String callSid) {
025            this.callSid = callSid;
026        }
027    
028        public String getConferenceSid() {
029            return this.conferenceSid;
030        }
031        public void setConferenceSid(String conferenceSid) {
032            this.conferenceSid = conferenceSid;
033        }
034    
035        public boolean isMuted() {
036            return this.muted;
037        }
038        public void setMuted(boolean muted) {
039            this.muted = muted;
040        }
041    
042        public boolean isStartConferenceOnEnter() {
043            return this.startConferenceOnEnter;
044        }
045        public void setStartConferenceOnEnter(boolean startConferenceOnEnter) {
046            this.startConferenceOnEnter = startConferenceOnEnter;
047        }
048    
049        public boolean isEndConferenceOnExit() {
050            return this.endConferenceOnExit;
051        }
052        public void setEndConferenceOnExit(boolean endConferenceOnExit) {
053            this.endConferenceOnExit = endConferenceOnExit;
054        }
055    }
