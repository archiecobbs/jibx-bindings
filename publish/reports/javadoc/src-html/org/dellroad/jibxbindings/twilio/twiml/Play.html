010    import java.net.URI;

012    /**
013     * TwiML <code>&lt;Play&gt;</code> verb.
014     */
015    public class Play implements GatherVerb {
016    
017        public static final int DEFAULT_LOOP = 1;
018    
019        private int loop = DEFAULT_LOOP;
020        private URI uri;
021

023            return this.loop;
024        }
025        public void setLoop(int loop) {
026            this.loop = loop;
027        }
028    
029        public URI getURI() {
030            return this.uri;
031        }
032        public void setURI(URI uri) {
033            this.uri = uri;
034        }
035    }
