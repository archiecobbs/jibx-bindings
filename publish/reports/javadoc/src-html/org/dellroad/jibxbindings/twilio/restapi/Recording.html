008    package org.dellroad.jibxbindings.twilio.restapi;
009    
010    /**
011     * Represents a <code>&lt;Recording&gt;</code>.
012     */
013    public class Recording extends SidAccountResource implements TwilioResponseContent {
014    
015        private String callSid;
016        private int duration;
017        private APIVersion apiVersion;
018    
019        public String getCallSid() {
020            return this.callSid;
021        }
022        public void setCallSid(String callSid) {
023            this.callSid = callSid;
024        }
025    
026        public int getDuration() {
027            return this.duration;
028        }
029        public void setDuration(int duration) {
030            this.duration = duration;
031        }
032    
033        public APIVersion getAPIVersion() {
034            return this.apiVersion;
035        }
036        public void setAPIVersion(APIVersion apiVersion) {
037            this.apiVersion = apiVersion;
038        }
039    }
