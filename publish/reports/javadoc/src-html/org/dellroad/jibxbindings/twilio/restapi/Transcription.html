008package org.dellroad.jibxbindings.twilio.restapi;
009
010/**
011 * Represents a <code>&lt;Transcription&gt;</code>.
012 */
013public class Transcription extends SidAccountResource implements TwilioResponseContent {
014
015    private TranscriptionStatus status;
016    private TranscriptionType type;
017    private String recordingSid;
018    private int duration;
019    private String transcriptionText;
020    private double price;
021    private String priceUnit;
022    private APIVersion apiVersion;
023
024    public TranscriptionStatus getStatus() {
025        return this.status;
026    }
027    public void setStatus(TranscriptionStatus status) {
028        this.status = status;
029    }
030
031    public TranscriptionType getType() {
032        return this.type;
033    }
034    public void setType(TranscriptionType type) {
035        this.type = type;
036    }
037
038    public String getRecordingSid() {
039        return this.recordingSid;
040    }
041    public void setRecordingSid(String recordingSid) {
042        this.recordingSid = recordingSid;
043    }
044
045    public int getDuration() {
046        return this.duration;
047    }
048    public void setDuration(int duration) {
049        this.duration = duration;
050    }
051
052    public String getTranscriptionText() {
053        return this.transcriptionText;
054    }
055    public void setTranscriptionText(String transcriptionText) {
056        this.transcriptionText = transcriptionText;
057    }
058
059    public double getPrice() {
060        return this.price;
061    }
062    public void setPrice(double price) {
063        this.price = price;
064    }
065
066    public String getPriceUnit() {
067        return this.priceUnit;
068    }
069    public void setPriceUnit(String priceUnit) {
070        this.priceUnit = priceUnit;
071    }
072
073    public APIVersion getAPIVersion() {
074        return this.apiVersion;
075    }
076    public void setAPIVersion(APIVersion apiVersion) {
077        this.apiVersion = apiVersion;
078    }
079}
