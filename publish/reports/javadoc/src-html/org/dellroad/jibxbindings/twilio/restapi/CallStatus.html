008    package org.dellroad.jibxbindings.twilio.restapi;
009    
010    import org.dellroad.jibxbindings.ParseUtil;
011    import org.dellroad.jibxbindings.XMLEnum;
012    import org.jibx.runtime.JiBXParseException;
013    
014    /**
015     * Call statuses.
016     */
017    public enum CallStatus implements XMLEnum {
018        QUEUED("queued"),
019        RINGING("ringing"),
020        IN_PROGRESS("in-progress"),
021        COMPLETED("completed"),
022        FAILED("failed"),
023        BUSY("busy"),
024        NO_ANSWER("no-answer");
025    
026        private final String xmlName;
027    
028        private CallStatus(String xmlName) {
029            this.xmlName = xmlName;
030        }
031    
032        @Override
033        public String getXMLName() {
034            return this.xmlName;
035        }
036    
037        public static CallStatus deserializeXML(String string) throws JiBXParseException {
038            return ParseUtil.deserializeXMLEnum(string, CallStatus.values());
039        }
040    }
