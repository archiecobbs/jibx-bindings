008    package org.dellroad.jibxbindings.twilio.twiml;
009    
010    import java.net.URI;
011    
012    import org.dellroad.jibxbindings.twilio.Method;
013    
014    /**
015     * TwiML <code>&lt;Record&gt;</code> verb.
016     */
017    public class Record implements Verb {
018    
019        public static final int DEFAULT_TIMEOUT = 5;
020        public static final int DEFAULT_MAX_LENGTH = 60 * 60;
021        public static final String DEFAULT_FINISH_ON_KEY = "1234567890*#";
022    
023        private URI action;
024        private Method method = Method.POST;
025        private int timeout = DEFAULT_TIMEOUT;
026        private String finishOnKey = DEFAULT_FINISH_ON_KEY;
027        private int maxLength = DEFAULT_MAX_LENGTH;
028        private boolean transcribe;
029        private URI transcribeCallback;
030        private boolean playBeep = true;
031    
032        public URI getAction() {
033            return this.action;
034        }
035        public void setAction(URI action) {
036            this.action = action;
037        }
038    
039        public Method getMethod() {
040            return this.method;
041        }
042        public void setMethod(Method method) {
043            this.method = method;
044        }
045    
046        public int getTimeout() {
047            return this.timeout;
048        }
049        public void setTimeout(int timeout) {
050            this.timeout = timeout;
051        }
052    
053        public String getFinishOnKey() {
054            return this.finishOnKey;
055        }
056        public void setFinishOnKey(String finishOnKey) {
057            this.finishOnKey = finishOnKey;
058        }
059    
060        public int getMaxLength() {
061            return this.maxLength;
062        }
063        public void setMaxLength(int maxLength) {
064            this.maxLength = maxLength;
065        }
066    
067        public boolean isTranscribe() {
068            return this.transcribe;
069        }
070        public void setTranscribe(boolean transcribe) {
071            this.transcribe = transcribe;
072        }
073    
074        public URI getTranscribeCallback() {
075            return this.transcribeCallback;
076        }
077        public void setTranscribeCallback(URI transcribeCallback) {
078            this.transcribeCallback = transcribeCallback;
079        }
080    
081        public boolean isPlayBeep() {
082            return this.playBeep;
083        }
084        public void setPlayBeep(boolean playBeep) {
085            this.playBeep = playBeep;
086        }
087    }
