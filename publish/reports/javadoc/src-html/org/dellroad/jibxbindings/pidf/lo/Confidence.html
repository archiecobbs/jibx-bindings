008package org.dellroad.jibxbindings.pidf.lo;
009
010import org.jibx.runtime.JiBXParseException;
011
012/**
013 * The {@code <con:confidence>} element as defined in RFC 4119.
014 *
015 * @see <a href="https://datatracker.ietf.org/doc/draft-ietf-geopriv-uncertainty/?include_text=1">Representation of Uncertainty and Confidence in PIDF-LO</a>
016 */
017public class Confidence {
018
019    public static final String UNKNOWN_VALUE = "unknown";
020
021    private ProbabilityDensityFunction probabilityDensityFunction = ProbabilityDensityFunction.UNKNOWN;
022    private float value = Float.NaN;
023
024    public ProbabilityDensityFunction getProbabilityDensityFunction() {
025        return this.probabilityDensityFunction;
026    }
027    public void setProbabilityDensityFunction(ProbabilityDensityFunction probabilityDensityFunction) {
028        this.probabilityDensityFunction = probabilityDensityFunction;
029    }
030
031    public float getValue() {
032        return this.value;
033    }
034    public void setValue(float value) {
035        this.value = value;
036    }
037
038// JiBX
039
040    public static String serializeConfidenceValue(float value) {
041        if (Float.isNaN(value) || Float.isInfinite(value))
042            return UNKNOWN_VALUE;
043        value = Math.min(Math.max(value, 0.0f), 100.0f);
044        return String.format("%.1f", value);
045    }
046
047    public static float deserializeConfidenceValue(String string) throws JiBXParseException {
048        if (string == null || (string = string.trim()).length() == 0 || string.equalsIgnoreCase(UNKNOWN_VALUE))
049            return Float.NaN;
050        float value;
051        try {
052            value = Float.parseFloat(string);
053        } catch (NumberFormatException e) {
054            throw new JiBXParseException("incorrectly formatted value", string, e);
055        }
056        value = Math.min(Math.max(value, 0.0f), 100.0f);
057        return value;
058    }
059
060    /**
061     * Indicates probability density function associated with a {@link Confidence} value.
062     */
063    public enum ProbabilityDensityFunction {
064        UNKNOWN("unknown"),
065        NORMAL("normal"),
066        RECTANGULAR("rectangular");
067
068        private final String xmlValue;
069
070        private ProbabilityDensityFunction(String xmlValue) {
071            this.xmlValue = xmlValue;
072        }
073
074        public final String getXMLValue() {
075            return this.xmlValue;
076        }
077    }
078}
