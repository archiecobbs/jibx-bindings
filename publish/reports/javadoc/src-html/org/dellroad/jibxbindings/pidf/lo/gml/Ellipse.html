008package org.dellroad.jibxbindings.pidf.lo.gml;
009
010/**
011 * GML ellipse.
012 */
013public class Ellipse extends Surface {
014
015    private double[] center;
016    private Measure semiMajorAxis;
017    private Measure semiMinorAxis;
018    private Measure orientation;
019
020    /**
021     * The center of the ellipse.
022     */
023    public double[] getCenter() {
024        return this.center;
025    }
026    public void setCenter(double[] center) {
027        this.center = center;
028    }
029
030    /**
031     * The ellipse's {@code <gs:semiMajorAxis>}.
032     */
033    public Measure getSemiMajorAxis() {
034        return this.semiMajorAxis;
035    }
036    public void setSemiMajorAxis(Measure semiMajorAxis) {
037        this.semiMajorAxis = semiMajorAxis;
038    }
039
040    /**
041     * The ellipse's {@code <gs:semiMinorAxis>}.
042     */
043    public Measure getSemiMinorAxis() {
044        return this.semiMinorAxis;
045    }
046    public void setSemiMinorAxis(Measure semiMinorAxis) {
047        this.semiMinorAxis = semiMinorAxis;
048    }
049
050    /**
051     * The ellipse's {@code <gs:orientation>}.
052     */
053    public Measure getOrientation() {
054        return this.orientation;
055    }
056    public void setOrientation(Measure orientation) {
057        this.orientation = orientation;
058    }
059
060    @Override
061    public void visit(GMLObjectSwitch gmlObjectSwitch) {
062        gmlObjectSwitch.caseEllipse(this);
063    }
064}
