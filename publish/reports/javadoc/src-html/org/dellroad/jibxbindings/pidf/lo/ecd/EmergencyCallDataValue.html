008package org.dellroad.jibxbindings.pidf.lo.ecd;
009
010import java.util.ArrayList;
011import java.util.List;
012
013/**
014 * The {@code <gp:EmergencyCallDataValue>} element.
015 */
016public class EmergencyCallDataValue {
017
018    private List<ProviderInfo> providerInfos = new ArrayList<>();
019    private List<ServiceInfo> serviceInfos = new ArrayList<>();
020    private List<DeviceInfo> deviceInfos = new ArrayList<>();
021    private List<SubscriberInfo> subscriberInfos = new ArrayList<>();
022    private List<Comment> comments = new ArrayList<>();
023
024    public List<ProviderInfo> getProviderInfos() {
025        return this.providerInfos;
026    }
027    public void setProviderInfos(List<ProviderInfo> providerInfos) {
028        this.providerInfos = providerInfos;
029    }
030
031    public List<ServiceInfo> getServiceInfos() {
032        return this.serviceInfos;
033    }
034    public void setServiceInfos(List<ServiceInfo> serviceInfos) {
035        this.serviceInfos = serviceInfos;
036    }
037
038    public List<DeviceInfo> getDeviceInfos() {
039        return this.deviceInfos;
040    }
041    public void setDeviceInfos(List<DeviceInfo> deviceInfos) {
042        this.deviceInfos = deviceInfos;
043    }
044
045    public List<SubscriberInfo> getSubscriberInfos() {
046        return this.subscriberInfos;
047    }
048    public void setSubscriberInfos(List<SubscriberInfo> subscriberInfos) {
049        this.subscriberInfos = subscriberInfos;
050    }
051
052    public List<Comment> getComments() {
053        return this.comments;
054    }
055    public void setComments(List<Comment> comments) {
056        this.comments = comments;
057    }
058}
