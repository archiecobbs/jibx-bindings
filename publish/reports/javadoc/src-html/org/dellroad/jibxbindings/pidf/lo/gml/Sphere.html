008package org.dellroad.jibxbindings.pidf.lo.gml;
009
010/**
011 * GML sphere.
012 */
013public class Sphere extends Solid {
014
015    private double[] center;
016    private Measure radius;
017
018    /**
019     * The center of the sphere.
020     */
021    public double[] getCenter() {
022        return this.center;
023    }
024    public void setCenter(double[] center) {
025        this.center = center;
026    }
027
028    /**
029     * The sphere's {@code <gs:radius>}.
030     */
031    public Measure getRadius() {
032        return this.radius;
033    }
034    public void setRadius(Measure radius) {
035        this.radius = radius;
036    }
037
038    @Override
039    public void visit(GMLObjectSwitch gmlObjectSwitch) {
040        gmlObjectSwitch.caseSphere(this);
041    }
042}
