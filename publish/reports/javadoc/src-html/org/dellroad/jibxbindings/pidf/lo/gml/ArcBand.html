008package org.dellroad.jibxbindings.pidf.lo.gml;
009
010/**
011 * GML arc band.
012 */
013public class ArcBand extends Surface {
014
015    private double[] center;
016    private Measure innerRadius;
017    private Measure outerRadius;
018    private Measure startAngle;
019    private Measure openingAngle;
020
021    /**
022     * The center of the arc band.
023     */
024    public double[] getCenter() {
025        return this.center;
026    }
027    public void setCenter(double[] center) {
028        this.center = center;
029    }
030
031    /**
032     * The {@code <gs:innerRadius>}.
033     */
034    public Measure getInnerRadius() {
035        return this.innerRadius;
036    }
037    public void setInnerRadius(Measure innerRadius) {
038        this.innerRadius = innerRadius;
039    }
040
041    /**
042     * The {@code <gs:outerRadius>}.
043     */
044    public Measure getOuterRadius() {
045        return this.outerRadius;
046    }
047    public void setOuterRadius(Measure outerRadius) {
048        this.outerRadius = outerRadius;
049    }
050
051    /**
052     * The {@code <gs:startAngle>}.
053     */
054    public Measure getStartAngle() {
055        return this.startAngle;
056    }
057    public void setStartAngle(Measure startAngle) {
058        this.startAngle = startAngle;
059    }
060
061    /**
062     * The {@code <gs:openingAngle>}.
063     */
064    public Measure getOpeningAngle() {
065        return this.openingAngle;
066    }
067    public void setOpeningAngle(Measure openingAngle) {
068        this.openingAngle = openingAngle;
069    }
070
071    @Override
072    public void visit(GMLObjectSwitch gmlObjectSwitch) {
073        gmlObjectSwitch.caseArcBand(this);
074    }
075}
